39// The RGB matrix provides the framebuffer and the facilities to constantly
40// update the LED matrix.
41//
42// This implement the Canvas interface that represents the display with
43// (led_cols * chained_displays)x(rows * parallel_displays) pixels.
44//
45// If can do multi-buffering using the CreateFrameCanvas() and SwapOnVSync()
46// methods. This is useful for animations and to prevent tearing.
47//
48// If you arrange the panels in a different way in the physical space, write
49// a CanvasTransformer that does coordinate remapping and which should be added
50// to the transformers, like with UArrangementTransformer in demo-main.cc.
51class RGBMatrix : public Canvas {
52public:

63 // Name of the hardware mapping. Something like "regular" or "adafruit-hat"

66 // The "rows" are the number
67 // of rows supported by the display, so 32 or 16. Default: 32.
68 // Flag: --led-rows
69 int rows;
70
71 // The "cols" are the number of columns per panel. Typically something
72 // like 32, but also 64 is possible. Sometimes even 40.
73 // cols * chain_length is the total length of the display, so you can
74 // represent a 64 wide display as cols=32, chain=2 or cols=64, chain=1;
75 // same thing, but more convenient to think of.
76 // Flag: --led-cols
77 int cols;
78
79 // The chain_length is the number of displays daisy-chained together
80 // (output of one connected to input of next). Default: 1
81 // Flag: --led-chain

83
84 // The number of parallel chains connected to the Pi; in old Pis with 26
85 // GPIO pins, that is 1, in newer Pis with 40 interfaces pins, that can
86 // also be 2 or 3. The effective number of pixels in vertical direction is
87 // then thus rows * parallel. Default: 1
88 // Flag: --led-parallel

170 // A factory that parses your main() commandline flags to read options
171 // meant to configure the the matrix and returns a freshly allocated matrix.
172 //
173 // Optionally, you can pass in option structs with a couple of defaults
174 // which are used unless overwritten on the command line.
175 // A matrix is created and returned; also the options structs are

194 // -- Canvas interface. These write to the active FrameCanvas
195 // (see documentation in canvas.h)
196 //
197 // Since this is updating the canvas that is currently displayed, this

207 // -- Double- and Multibuffering.
208
209 // Create a new buffer to be used for multi-buffering. The returned new
210 // Buffer implements a Canvas with the same size of thie RGBMatrix.
211 // You can use it to draw off-screen on it, then swap it with the active
212 // buffer using SwapOnVSync(). That would be classic double-buffering.
213 //
214 // You can also create as many FrameCanvas as you like and for instance use
215 // them to pre-fill scenes of an animation for fast playback later.
216 //
217 // The ownership of the created Canvases remains with the RGBMatrix, so you
218 // don't have to worry about deleting them (but you also don't want to create
219 // more than needed as this will fill up your memory as they are only deleted
220 // when the RGBMatrix is deleted).
221 FrameCanvas *CreateFrameCanvas();
222
223 // This method waits to the next VSync and swaps the active buffer with the
224 // supplied buffer. The formerly active buffer is returned.
225 //
226 // If you pass in NULL, the active buffer is returned, but it won't be
227 // replaced with NULL. You can use the NULL-behavior to just wait on
228 // VSync or to retrieve the initial buffer when preparing a multi-buffer
229 // animation.
230 //
231 // The optional "framerate_fraction" parameter allows to choose which
232 // multiple of the global frame-count to use. So it slows down your animation
233 // to an exact integer fraction of the refresh rate.
234 // Default is 1, so immediately next available frame.
235 // (Say you have 140Hz refresh rate, then a value of 5 would give you an
236 // 28Hz animation, nicely locked to the refresh-rate).
237 // If you combine this with Options::limit_refresh_rate_hz you can create
238 // time-correct animations.
239 FrameCanvas *SwapOnVSync(FrameCanvas *other, unsigned framerate_fraction = 1);
240
241 // -- Setting shape and behavior of matrix.
242
243 // Apply a pixel mapper. This is used to re-map pixels according to some
244 // scheme implemented by the PixelMapper. Does _not_ take ownership of the
245 // mapper. Mapper can be NULL, in which case nothing happens.
246 // Returns a boolean indicating if this was successful.
247 bool ApplyPixelMapper(const PixelMapper *mapper);
248
249 // Note, there used to be ApplyStaticTransformer(), which has been deprecated
250 // since 2018 and changed to a compile-time option, then finally removed
251 // in 2020. Use PixelMapper instead, which is simpler and more intuitive.
252
253 // Set PWM bits used for output. Default is 11, but if you only deal with
254 // limited comic-colors, 1 might be sufficient. Lower require less CPU and
255 // increases refresh-rate.
256 //
257 // Returns boolean to signify if value was within range.
258 //
259 // This sets the PWM bits for the current active FrameCanvas and future
260 // ones that are created with CreateFrameCanvas().
261 bool SetPWMBits(uint8_t value);
262 uint8_t pwmbits(); // return the pwm-bits of the currently active buffer.
263
264 // Map brightness of output linearly to input with CIE1931 profile.
265 void set_luminance_correct(bool on);
266 bool luminance_correct() const;

268 // Set brightness in percent for all created FrameCanvas. 1%..100%.
269 // This will only affect newly set pixels.
270 void SetBrightness(uint8_t brightness);
271 uint8_t brightness();

273 //-- GPIO interaction.
274 // This library uses the GPIO pins to drive the matrix; this is a safe way
275 // to request the 'remaining' bits to be used for user purposes.
276
277 // Request user readable GPIO bits.
278 // This function allows you to request pins you'd like to read with
279 // AwaitInputChange().
280 // Only bits that are not already in use for reading or wrtiting
281 // by the matrix are allowed.
282 // Input is a bitmap of all the GPIO bits you're interested in; returns all
283 // the bits that are actually available.
284 uint64_t RequestInputs(uint64_t all_interested_bits);
285
286 // This function will return whenever the GPIO input pins
287 // change (pins that are not already in use for output, that is) or the
288 // timeout is reached. You need to have reserved the inputs with
289 // matrix->RequestInputs(...) first (e.g.
290 // matrix->RequestInputs((1<<25)|(1<<24));
291 //
292 // A positive timeout waits the given amount of milliseconds for a change
293 // (e.g. a button-press) to occur; if there is no change, it will just
294 // return the last value.
295 // If you just want to know how the pins are right now, call with zero
296 // timeout.
297 // A negative number waits forever and will only return if there is a change.
298 //
299 // This function only samples between display refreshes so polling some
300 // input does not generate flicker and provide a convenient change interface.
301 //
302 // Returns the bitmap of all GPIO input pins.
303 uint64_t AwaitInputChange(int timeout_ms);
304
305 // Request user writable GPIO bits.
306 // This allows to request a bitmap of GPIO-bits to be used by the user for
307 // writing.
308 // Only bits that are not already in use for reading or wrtiting
309 // by the matrix are allowed.
310 // Returns the subset bits that are _actually_ available,
311 uint64_t RequestOutputs(uint64_t output_bits);
312
313 // Set the user-settable bits according to output bits.
314 void OutputGPIO(uint64_t output_bits);
315
316 // Legacy way to set gpio pins. We're not doing this anymore but need to
317 // be source-compatible with old calls of the form
318 // matrix->gpio()->RequestInputs(...)
319 //
320 // Don't use, use AwaitInputChange() directly.
321 RGBMatrix *gpio() __attribute__((deprecated)) { return this; }
322
323 //-- Rarely needed
324 // Start the refresh thread.
325 // This is only needed if you chose RuntimeOptions::daemon = -1 (see below),
326 // otherwise the refresh thread is already started.
327 bool StartRefresh();

400// Runtime options to simplify doing common things for many programs such as
401// dropping privileges and becoming a daemon.
402struct RuntimeOptions {
403 RuntimeOptions();
404
405 int gpio_slowdown; // 0 = no slowdown. Flag: --led-slowdown-gpio
406
407 // ----------
408 // If the following options are set to disabled with -1, they are not
409 // even offered via the command line flags.
410 // ----------
411
412 // Thre are three possible values here
413 // -1 : don't leave choise of becoming daemon to the command line
414 // parsing. If set to -1, the --led-daemon option is not offered.
415 // 0 : do not becoma a daemon, run in forgreound (default value)
416 // 1 : become a daemon, run in background.
417 //
418 // If daemon is disabled (= -1), the user has to call
419 // RGBMatrix::StartRefresh() manually once the matrix is created, to leave
420 // the decision to become a daemon
421 // after the call (which requires that no threads have been started yet).
422 // In the other cases (off or on), the choice is already made, so the
423 // thread is conveniently already started for you.
424 int daemon; // -1 disabled. 0=off, 1=on. Flag: --led-daemon
425
426 // Drop privileges from 'root' to drop_priv_user/group once the hardware is
427 // initialized.
428 // This is usually a good idea unless you need to stay on elevated privs.
429 // -1, 0, 1 similar meaning to 'daemon' above.
430 int drop_privileges; // -1 disabled. 0=off, 1=on. flag: --led-drop-privs
431
432 // By default, the gpio is initialized for you, but if you run on a platform
433 // not the Raspberry Pi, this will fail. If you don't need to access GPIO
434 // e.g. you want to just create a stream output (see content-streamer.h),
435 // set this to false.
436 bool do_gpio_init;
437
438 // If drop privileges is enabled, this is the user/group we drop privileges
439 // to. Unless chosen otherwise, the default is "daemon" for user and group.
440 const char *drop_priv_user;
441 const char *drop_priv_group;
442};
