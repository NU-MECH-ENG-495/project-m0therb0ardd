32// For now, everything is initialized as output.
33class GPIO {
34public:
35 GPIO();
36
37 // Initialize before use. Returns 'true' if successful, 'false' otherwise
38 // (e.g. due to a permission problem).
39 bool Init(int slowdown);
40
41 // Initialize outputs.
42 // Returns the bits that were available and could be set for output.
43 // (never use the optional adafruit_hack_needed parameter, it is used
44 // internally to this library).
45 gpio_bits_t InitOutputs(gpio_bits_t outputs,
46 bool adafruit_hack_needed = false);
47
48 // Request given bitmap of GPIO inputs.
49 // Returns the bits that were available and could be reserved.
50 gpio_bits_t RequestInputs(gpio_bits_t inputs);
51
52 // Set the bits that are '1' in the output. Leave the rest untouched.
53 inline void SetBits(gpio_bits_t value) {
54 if (!value) return;
55 WriteSetBits(value);
56 delay();
57 }
58
59 // Clear the bits that are '1' in the output. Leave the rest untouched.
60 inline void ClearBits(gpio_bits_t value) {
61 if (!value) return;
62 WriteClrBits(value);
63 delay();
64 }
65
66 // Write all the bits of "value" mentioned in "mask". Leave the rest untouched.
67 inline void WriteMaskedBits(gpio_bits_t value, gpio_bits_t mask) {
68 // Writing a word is two operations. The IO is actually pretty slow, so
69 // this should probably be unnoticable.
70 WriteClrBits(~value & mask);
71 WriteSetBits(value & mask);
72 delay();
73 }
74
75 inline gpio_bits_t Read() const { return ReadRegisters() & input_bits_; }
76
77 // Return if this is appears to be a Pi4
78 static bool IsPi4();
79
80private:
81 inline void delay() const {
82#if LED_MATRIX_ALLOW_BARRIER_DELAY
83 if (slowdown_ == -1) {
84 asm volatile("dsb\tst");
85 return;
86 }
87#endif
88 for (int n = 0; n < slowdown_; n++) {
89 *gpio_clr_bits_low_ = 0;
90 }
91 }
92
93 inline gpio_bits_t ReadRegisters() const {
94 return (static_cast<gpio_bits_t>(*gpio_read_bits_low_)
95#ifdef ENABLE_WIDE_GPIO_COMPUTE_MODULE
96 | (static_cast<gpio_bits_t>(*gpio_read_bits_low_) << 32)
97#endif
98 );
99 }
100
101 inline void WriteSetBits(gpio_bits_t value) {
102 *gpio_set_bits_low_ = static_cast<uint32_t>(value & 0xFFFFFFFF);
103#ifdef ENABLE_WIDE_GPIO_COMPUTE_MODULE
104 if (uses_64_bit_)
105 *gpio_set_bits_high_ = static_cast<uint32_t>(value >> 32);
106#endif
107 }
108
109 inline void WriteClrBits(gpio_bits_t value) {
110 *gpio_clr_bits_low_ = static_cast<uint32_t>(value & 0xFFFFFFFF);
111#ifdef ENABLE_WIDE_GPIO_COMPUTE_MODULE
112 if (uses_64_bit_)
113 *gpio_clr_bits_high_ = static_cast<uint32_t>(value >> 32);
114#endif
115 }
116
117private:
118 gpio_bits_t output_bits_;
119 gpio_bits_t input_bits_;
120 gpio_bits_t reserved_bits_;
121 int slowdown_;
122
123 volatile uint32_t *gpio_set_bits_low_;
124 volatile uint32_t *gpio_clr_bits_low_;
125 volatile uint32_t *gpio_read_bits_low_;
126
127#ifdef ENABLE_WIDE_GPIO_COMPUTE_MODULE
128 bool uses_64_bit_;
129 volatile uint32_t *gpio_set_bits_high_;
130 volatile uint32_t *gpio_clr_bits_high_;
131 volatile uint32_t *gpio_read_bits_high_;
132#endif
133};

135// A PinPulser is a utility class that pulses a GPIO pin. There can be various
136// implementations.
137class PinPulser {
138public:
139 // Factory for a PinPulser. Chooses the right implementation depending
140 // on the context (CPU and which pins are affected).
141 // "gpio_mask" is the mask that should be output (since we only
142 // need negative pulses, this is what it does)
143 // "nano_wait_spec" contains a list of time periods we'd like
144 // invoke later. This can be used to pre-process timings if needed.
145 static PinPulser *Create(GPIO *io, gpio_bits_t gpio_mask,
146 bool allow_hardware_pulsing,
147 const std::vector<int> &nano_wait_spec);
148
149 virtual ~PinPulser() {}
150
151 // Send a pulse with a given length (index into nano_wait_spec array).
152 virtual void SendPulse(int time_spec_number) = 0;
153
154 // If SendPulse() is asynchronously implemented, wait for pulse to finish.
155 virtual void WaitPulseFinished() {}
156};
